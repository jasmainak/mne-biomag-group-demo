scale = 1e6
plt.figure(figsize=(7, 5))
plt.plot(evoked.times * 1000, mapping['Scrambled'].data[idx] * scale,
         'r', linewidth=2, label='Scrambled')

         'g', linewidth=2, label='Unfamiliar')

         linewidth=2, label='Famous')

ax = plt.gca()
